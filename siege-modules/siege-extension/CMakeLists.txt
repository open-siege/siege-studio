cmake_minimum_required(VERSION 3.30)
project(siege-extension)

if (WIN32)
    add_library(detours STATIC IMPORTED)   
    set_target_properties(detours PROPERTIES IMPORTED_LOCATION ${CMAKE_CURRENT_SOURCE_DIR}/detours/detours.lib INTERFACE_INCLUDE_DIRECTORIES ${CMAKE_CURRENT_SOURCE_DIR}/detours)
    add_library(siege-extension STATIC src/input-filter.cpp)
    set_target_properties(siege-extension PROPERTIES CXX_STANDARD 23)
    target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/src)
    target_link_libraries(siege-extension PUBLIC siege-win-desktop detours)

    add_library(siege-extension-input-filter-raw-input SHARED src/input-filter.cpp src/input-filter-raw-input.cpp src/filter.def)
    set_target_properties(siege-extension-input-filter-raw-input PROPERTIES CXX_STANDARD 23)
    target_link_libraries(siege-extension-input-filter-raw-input PRIVATE siege-win-desktop detours)

    if (IS_DIRECTORY ${CMAKE_SOURCE_DIR}/libzt)
        add_library(wsock32-on-zero-tier SHARED src/wsock32-on-zero-tier.cpp src/wsock32-on-zero-tier.def)
        target_include_directories(wsock32-on-zero-tier PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src ${CMAKE_SOURCE_DIR}/libzt/include)
        set_target_properties(wsock32-on-zero-tier PROPERTIES CXX_STANDARD 23)
        target_link_libraries(wsock32-on-zero-tier PRIVATE siege-win-core)

        add_library(ws2_32-on-zero-tier SHARED src/wsock32-on-zero-tier.cpp src/ws2_32-on-zero-tier.def)
        target_include_directories(ws2_32-on-zero-tier PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src ${CMAKE_SOURCE_DIR}/libzt/include)
        set_target_properties(ws2_32-on-zero-tier PROPERTIES CXX_STANDARD 23)
        target_link_libraries(ws2_32-on-zero-tier PRIVATE siege-win-core)
    endif()
endif()
